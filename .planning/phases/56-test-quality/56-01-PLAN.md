**Step 1: Extend `src/e2e/tests/settings.page.ts`** to add helper methods for the Sonarr connection test flow.

Add these methods to the `SettingsPage` class:

```typescript
async enableSonarr(): Promise<void> {
    // Enable Sonarr integration via config API (same pattern as setup_seedsyncarr.sh)
    await this.page.request.get(
        'http://myapp:8800/server/config/set/sonarr/enabled/true'
    );
}

async setSonarrUrl(url: string): Promise<void> {
    await this.page.request.get(
        `http://myapp:8800/server/config/set/sonarr/sonarr_url/${encodeURIComponent(url)}`
    );
}

async setSonarrApiKey(key: string): Promise<void> {
    await this.page.request.get(
        `http://myapp:8800/server/config/set/sonarr/sonarr_api_key/${encodeURIComponent(key)}`
    );
}

async clickTestSonarrConnection(): Promise<void> {
    // The "Test Connection" button in the Sonarr fieldset
    const sonarrFieldset = this.page.locator('fieldset').filter({ has: this.page.locator('text=Sonarr URL') });
    await sonarrFieldset.locator('button:has-text("Test Connection")').click();
}

async getSonarrTestResult(): Promise<{ text: string; isDanger: boolean }> {
    const sonarrFieldset = this.page.locator('fieldset').filter({ has: this.page.locator('text=Sonarr URL') });
    const result = sonarrFieldset.locator('.test-result');
    await result.waitFor({ state: 'visible', timeout: 15000 });
    const text = await result.innerText();
    const isDanger = await result.evaluate(el => el.classList.contains('text-danger'));
    return { text, isDanger };
}

async disableSonarr(): Promise<void> {
    await this.page.request.get(
        'http://myapp:8800/server/config/set/sonarr/enabled/false'
    );
}
```

**Step 2: Create `src/e2e/tests/settings-error.spec.ts`** as a new spec file:

```typescript
import { test, expect } from '@playwright/test';
import { SettingsPage } from './settings.page';

test.describe('Settings page error states', () => {
    let settingsPage: SettingsPage;

    test.beforeEach(async ({ page }) => {
        settingsPage = new SettingsPage(page);
    });

    test.afterEach(async () => {
        // Restore: disable Sonarr so subsequent tests see clean state
        await settingsPage.disableSonarr();
    });

    test('should show error when Sonarr connection fails', async ({ page }) => {
        // Configure Sonarr with an unreachable URL via the config API
        await settingsPage.enableSonarr();
        await settingsPage.setSonarrUrl('http://nonexistent.invalid:8989');
        await settingsPage.setSonarrApiKey('fake-api-key-for-testing');

        // Navigate to Settings page (reload to pick up config changes)
        await settingsPage.navigateTo();

        // Click the "Test Connection" button in the Sonarr section
        await settingsPage.clickTestSonarrConnection();

        // Assert the error result is visible with text-danger class
        const result = await settingsPage.getSonarrTestResult();
        expect(result.isDanger).toBe(true);
        expect(result.text.length).toBeGreaterThan(0);
    });
});
```

**Key design decisions:**
- Uses the config API directly (same pattern as `setup_seedsyncarr.sh`) to set Sonarr to an unreachable URL before navigating
- Clicks the real "Test Connection" button in the UI, which triggers `onTestSonarrConnection()` in the Angular component
- Asserts the `.test-result` div appears with `text-danger` class (the Angular template applies `[class.text-danger]="!testSonarrConnectionResult.success"`)
- `afterEach` disables Sonarr to prevent state leakage to other specs
- Separate spec file (`settings-error.spec.ts`) isolates unhappy-path from the existing `settings.page.spec.ts` happy-path tests
- `fullyParallel: false` and `workers: 1` in playwright.config.ts ensure sequential execution, so this won't race with other specs
