function setInputsAndDetectChanges(
    visibleFiles: List<ViewFile>,
    selectedFiles: Set<string>
): void {
    const changes: SimpleChanges = {};
    if (component.visibleFiles !== visibleFiles) {
        changes["visibleFiles"] = new SimpleChange(component.visibleFiles, visibleFiles, false);
        component.visibleFiles = visibleFiles;
    }
    if (component.selectedFiles !== selectedFiles) {
        changes["selectedFiles"] = new SimpleChange(component.selectedFiles, selectedFiles, false);
        component.selectedFiles = selectedFiles;
    }
    if (Object.keys(changes).length > 0) {
        component.ngOnChanges(changes);
    }
    fixture.detectChanges();
}

    describe("FIX-01 DELETED union regression", () => {
        // Characterization tests per 76-CONTEXT.md D-01.
        // These document the v1.1.0 regression: when a selection includes a
        // DELETED file, the Queue action (dispatched against DELETED rows =
        // "Re-Queue from Remote") must remain enabled and onQueueClick()
        // must emit the DELETED filename.
        //
        // Wave 1 commits these red. Wave 2 drives them green with a
        // minimal edit (root cause TBD per D-02). Wave 3 adds the three
        // D-09 mixed-selection coverage tests alongside these.

        it("alone-DELETED: Queue enabled + onQueueClick emits deleted name (FIX-01)", () => {
            const deletedFile = new ViewFile({
                name: "deleted-file.mkv",
                status: ViewFile.Status.DELETED,
                remoteSize: 1024,
                localSize: 0,
                isQueueable: true,
                isStoppable: false,
                isExtractable: false,
                isArchive: false,
                isLocallyDeletable: false,
                isRemotelyDeletable: true,
            });
            const visible = List([deletedFile]);
            const selected = new Set<string>(["deleted-file.mkv"]);

            setInputsAndDetectChanges(visible, selected);

            expect(component.actionCounts.queueable)
                .withContext("DELETED row with remoteSize > 0 must be queueable")
                .toBeGreaterThanOrEqual(1);

            spyOn(component.queueAction, "emit");
            component.onQueueClick();

            expect(component.queueAction.emit).toHaveBeenCalledWith(
                jasmine.arrayContaining(["deleted-file.mkv"])
            );
        });

        it("mixed DELETED + DOWNLOADING: union includes Queue for DELETED row (FIX-01)", () => {
            const deletedFile = new ViewFile({
                name: "deleted-file.mkv",
                status: ViewFile.Status.DELETED,
                remoteSize: 1024,
                localSize: 0,
                isQueueable: true,
                isStoppable: false,
                isExtractable: false,
                isArchive: false,
                isLocallyDeletable: false,
                isRemotelyDeletable: true,
            });
            const downloadingFile = new ViewFile({
                name: "downloading-file.mkv",
                status: ViewFile.Status.DOWNLOADING,
                remoteSize: 2048,
                localSize: 512,
                isQueueable: false,
                isStoppable: true,
                isExtractable: false,
                isArchive: false,
                isLocallyDeletable: false,
                isRemotelyDeletable: true,
            });
            const visible = List([deletedFile, downloadingFile]);
            const selected = new Set<string>([
                "deleted-file.mkv",
                "downloading-file.mkv",
            ]);

            setInputsAndDetectChanges(visible, selected);

            expect(component.actionCounts.queueable)
                .withContext("Union: Queue enabled because DELETED row is queueable")

            component.onQueueClick();

            expect(component.queueAction.emit).toHaveBeenCalledWith(
                jasmine.arrayContaining(["deleted-file.mkv"])
            );
        });
    });
