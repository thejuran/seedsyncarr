```python
    name_to_root = {
        "showdir": "ShowDir",

    result = self.manager.process(name_to_root)

    root_name, matched_name = result[0]
    self.assertEqual("ShowDir", root_name)
    self.assertEqual("Episode.S01E01.mkv", matched_name)
    self.assertNotEqual(root_name, matched_name)
```
