```typescript
import { test as base, expect, type Page } from '@playwright/test';

export interface CspViolation {
    source: 'console' | 'event';
    blockedURI?: string;
    violatedDirective?: string;
    originalPolicy?: string;
    sourceFile?: string;
    lineNumber?: number;
    sample?: string;
    text?: string;  // raw console text for 'console' source
}

type Fixtures = {
    cspViolations: CspViolation[];
    allowViolations: boolean;
};

export const test = base.extend<Fixtures>({
    allowViolations: [false, { option: true }],

    cspViolations: async ({}, use) => {
        await use([]);
    },

    page: async ({ page, cspViolations, allowViolations }, use) => {
        // LOAD-BEARING ORDER (per RESEARCH §9 R-2 + §5):
        // 1. exposeFunction FIRST — survives navigations, creates the Node-side bridge
        //    before any init script can reference it.
        await page.exposeFunction('__reportCspViolation', (v: CspViolation) => {
            cspViolations.push({ ...v, source: 'event' });
        });

        // 2. addInitScript SECOND — re-runs on every navigation; registers the DOM
        //    listener fresh each time, but it calls the function exposed in step 1
        //    which persists.
        await page.addInitScript(() => {
            document.addEventListener('securitypolicyviolation', (e) => {
                // @ts-ignore — exposed function on window at runtime
                window.__reportCspViolation({
                    blockedURI: e.blockedURI,
                    violatedDirective: e.violatedDirective,
                    originalPolicy: e.originalPolicy,
                    sourceFile: e.sourceFile,
                    lineNumber: e.lineNumber,
                    sample: e.sample,
                });
            });
        });

        // 3. page.on('console') THIRD — defense-in-depth; catches violations that
        //    might slip past the DOM-event path (e.g. browser updates that change
        //    the event payload). Invariant substring per RESEARCH §4.
        page.on('console', (msg) => {
            const text = msg.text();
            if (msg.type() === 'error' &&
                text.includes('violates the following Content Security Policy directive')) {
                cspViolations.push({ source: 'console', text });
            }
        });

        // 4. Hand the page to the test body.
        await use(page);

        // 5. Teardown assertion — unless the spec opted out via test.use({ allowViolations: true }).
        if (!allowViolations) {
            expect(cspViolations, 'CSP violations detected').toEqual([]);
        }
    },
});

export { expect };
```

```typescript
test.describe('CSP violation canary', () => {
    test('Seeded inline-script injection fires a CSP violation', async ({ page, cspViolations }) => {
        // Inject after DOMContentLoaded (cross-document document.write is blocked
        // post-DCL by the Chromium intervention; appendChild is reliable).
        // The inline body has no matching sha256 hash in Angular autoCsp's
        // script-src, so 'strict-dynamic' + hash layer blocks it. Per RESEARCH §7.
        await page.addInitScript(() => {
            document.addEventListener('DOMContentLoaded', () => {
                const el = document.createElement('script');
                el.textContent = 'window.__canaryRan = true;';
                document.body.appendChild(el);
            });
        });

        await page.goto('/');

        // Either source (console or event) is sufficient — belt-and-braces.
        await expect.poll(() => cspViolations.length).toBeGreaterThan(0);

        const sawScriptSrc = cspViolations.some((v) =>
            (v.source === 'event' && v.violatedDirective?.startsWith('script-src')) ||
            (v.source === 'console' && v.text?.includes('script-src'))
        );
        expect(sawScriptSrc).toBe(true);
    });
});
```
