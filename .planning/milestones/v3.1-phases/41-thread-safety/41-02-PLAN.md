Fix three thread-safety issues in dispatch.py:

**THRD-03: Queue iteration under mutex**

Python's `queue.Queue` has a `.mutex` attribute (a threading.Lock) that protects its internal state. Direct access to `.queue` (the internal deque) without holding `.mutex` is a data race.

Three methods access `.queue` directly:
1. `status()` (line ~101): `list(self.__task_queue.queue)` -- read
2. `extract()` (line ~113): `for task in self.__task_queue.queue` -- read
3. `__worker()` (line ~165): `len(self.__task_queue.queue)` and `self.__task_queue.queue[0]` -- read

Fix each using copy-under-lock pattern with `self.__task_queue.mutex`:

**status() method** -- replace direct queue access:
```python
def status(self) -> List[ExtractStatus]:
    with self.__task_queue.mutex:
        tasks = list(self.__task_queue.queue)
    statuses = []
    for task in tasks:
        status = ExtractStatus(name=task.root_name,
                               is_dir=task.root_is_dir,
                               state=ExtractStatus.State.EXTRACTING)
        statuses.append(status)
    return statuses
```

**extract() method** -- replace the duplicate-check iteration:
```python
def extract(self, model_file: ModelFile):
    self.logger.debug("Received extract for {}".format(model_file.name))

    with self.__task_queue.mutex:
        for task in self.__task_queue.queue:
            if task.root_name == model_file.name:
                self.logger.info("Ignoring extract for {}, already exists".format(model_file.name))
                return
    # ... rest of method unchanged (task_queue.put is already thread-safe)
```

Note: The duplicate check reads the queue under mutex, and `self.__task_queue.put(task)` at the end is already thread-safe (Queue's put() acquires its own mutex internally). There is a TOCTOU window between the duplicate check and the put(), but this is acceptable: the worst case is two extract tasks for the same file, which is the same behavior as before this fix. The critical fix is preventing iteration over a concurrently-modified deque.

**__worker() method** -- replace direct `.queue` access in the inner while loop:
```python
def __worker(self):
    self.logger.debug("Started worker thread")

    while not self.__worker_shutdown.is_set():
        # Copy queue snapshot under mutex to check for work
        with self.__task_queue.mutex:
            has_tasks = len(self.__task_queue.queue) > 0

        while has_tasks and not self.__worker_shutdown.is_set():
            # Peek at first task under mutex
            with self.__task_queue.mutex:
                if len(self.__task_queue.queue) == 0:
                    break
                task = self.__task_queue.queue[0]

            # We have a task, extract archives one by one
            completed = True

            try:
                for archive_path, out_dir_path in task.archive_paths:
                    if self.__worker_shutdown.is_set():
                        self.logger.warning("Extraction failed, shutdown requested")
                        completed = False
                        break

                    self.logger.debug("Extracting {}".format(archive_path))
                    Extract.extract_archive(
                        archive_path=archive_path,
                        out_dir_path=out_dir_path
                    )

            except ExtractError:
                self.logger.exception("Caught an extraction error")
                completed = False
            finally:
                # pop the task (thread-safe Queue.get)
                self.__task_queue.get(block=False)

            # Send notification to listeners (copy-under-lock, see THRD-04 below)
            with self.__listeners_lock:
                listeners_snapshot = list(self.__listeners)
            for listener in listeners_snapshot:
                if completed:
                    listener.extract_completed(task.root_name, task.root_is_dir)
                else:
                    listener.extract_failed(task.root_name, task.root_is_dir)

            # Re-check queue under mutex for next iteration
            with self.__task_queue.mutex:
                has_tasks = len(self.__task_queue.queue) > 0

        time.sleep(ExtractDispatch.__WORKER_SLEEP_INTERVAL_IN_SECS)

    self.logger.debug("Stopped worker thread")
```

**THRD-04: Context manager lock + copy-under-lock for listeners**

In the __worker method's listener notification section, replace:
```python
self.__listeners_lock.acquire()
for listener in self.__listeners:
    ...
self.__listeners_lock.release()
```

With copy-under-lock using context manager:
```python
with self.__listeners_lock:
    listeners_snapshot = list(self.__listeners)
for listener in listeners_snapshot:
    ...
```

This change is already shown in the __worker refactor above.

Also update the `add_listener` method to use context manager style:
```python
def add_listener(self, listener: ExtractListener):
    with self.__listeners_lock:
        self.__listeners.append(listener)
```

Instead of the current acquire/release pattern.

def test_status_returns_consistent_snapshot(self):
    """Verify status() returns a consistent snapshot even with concurrent modifications."""
    self.mock_is_archive.return_value = True

    # Create files and add to queue
    files = []
    for i in range(10):
        mf = ModelFile("file{}".format(i), False)
        mf.local_size = 100
        files.append(mf)

    # Block extraction so tasks stay in queue
    barrier = threading.Event()
    def _block_extract(**kwargs):
        barrier.wait(timeout=5)
    self.mock_extract_archive.side_effect = _block_extract

    for mf in files:
        self.dispatch.extract(mf)

    # Call status multiple times - should not raise or return inconsistent data
    for _ in range(20):
        status = self.dispatch.status()
        # Status should have a consistent count (may decrease as worker processes)
        self.assertIsInstance(status, list)
        for s in status:
            self.assertIsInstance(s, ExtractStatus)

    barrier.set()
```

def test_listener_notification_allows_concurrent_add(self):
    """Verify listener notification uses copy-under-lock so adding during notify is safe."""
    self.mock_is_archive.return_value = True

    second_listener = DummyExtractListener()
    second_listener.extract_completed = MagicMock()
    second_listener.extract_failed = MagicMock()

    def _on_complete(name, is_dir):
        # Add another listener during notification -- would fail without copy-under-lock
        self.dispatch.add_listener(second_listener)
    self.listener.extract_completed = MagicMock(side_effect=_on_complete)

    mf = ModelFile("aaa", False)
    mf.local_size = 100

    self.dispatch.add_listener(self.listener)
    self.dispatch.extract(mf)

    # Wait for completion
    while self.listener.extract_completed.call_count < 1:
        pass
    time.sleep(0.1)

    self.listener.extract_completed.assert_called_once_with("aaa", False)
    # Second listener was added during notification, should not have been called for this event
    second_listener.extract_completed.assert_not_called()
```
